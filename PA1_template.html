## [1] 2304

## [1] 0.1311475

near 13% is missing

for this task i create a separate dataframe “datawithoutNA” in which i fill NA values using the mean for that 5-minute interval 

datawithoutNA <- data

for (i in 1:nrow(datawithoutNA)) {
        if (is.na(datawithoutNA$steps[i]) == TRUE) {
                datawithoutNA$steps[i] <- dailypattern$steps[dailypattern$interval == datawithoutNA$interval[i]]
        }
}
